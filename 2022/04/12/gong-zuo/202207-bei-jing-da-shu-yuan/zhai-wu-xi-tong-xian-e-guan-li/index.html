新增债务限额

列表查询

SELECT T.FISCAL_YEAR,
       AD.NAME                              AS                                     AD_NAME,
       T.BATCH_NO                           AS                                     BATCH_NO,
       (NVL(T.NEW_GENDEBT_LIMIT, 0) / 10000 + NVL(T.NEW_SPEDEBT_LIMIT, 0) / 10000) HJ,
       NVL(T.NEW_GENDEBT_LIMIT, 0) / 10000  AS                                     NEW_GENDEBT_LIMIT,
       NVL(T.FOREIGN_DEBT_LIMIT, 0) / 10000 AS                                     FOREIGN_DEBT_LIMIT,
       NVL(T.NEW_SPEDEBT_LIMIT, 0) / 10000  AS                                     NEW_SPEDEBT_LIMIT
FROM BAS_DEBT_V_LIMIT T
         LEFT JOIN DSY_V_ELE_AD AD ON T.REC_AD_CODE = AD.CODE
WHERE T.BILL_TYPE = '01'
  AND T.DATA_TYPE = '01'
  AND T.REC_AD_CODE = 42
  AND T.FISCAL_YEAR = 2022
ORDER BY T.REC_DATE DESC, T.FISCAL_YEAR DESC, T.BATCH_NO DESC;

限额分解录入

列表查询

SELECT T.DEBT_LIMIT_ID,
       T.FISCAL_YEAR,
       AD.NAME                                                      AD_NAME,
       T.BATCH_NO,
       NVL(T.NEW_GENDEBT_LIMIT, 0) + NVL(T.NEW_SPEDEBT_LIMIT, 0) AS NEW_DEBT_LIMIT,
       NVL(T.NEW_GENDEBT_LIMIT, 0)                               AS NEW_GENDEBT_LIMIT,
       NVL(T.FOREIGN_DEBT_LIMIT, 0)                              AS FOREIGN_DEBT_LIMIT,
       NVL(T.NEW_SPEDEBT_LIMIT, 0)                               AS NEW_SPEDEBT_LIMIT
FROM BAS_DEBT_V_LIMIT T
         LEFT JOIN DSY_V_ELE_AD AD ON T.REC_AD_CODE = AD.CODE
WHERE T.BILL_TYPE = '02'
  AND T.DATA_TYPE = '01'
  AND T.IS_DELETED = '2'
  and T.ad_code like concat(42, '%')
  AND T.ad_code like concat(42, '%')
  and exists(select 1
             from dsy_t_wf_task_complete wf
             where wf.wf_id = concat(T.wf_id, '')
               and wf.node_current_id = 1
               and wf.ywsj_id = T.DEBT_LIMIT_ID)
  and T.WF_ID = 100110
  and (T.NODE_CURRENT_ID != 1 or (T.NODE_CURRENT_ID = 1 and T.WF_STATUS != '001'))
  AND T.FISCAL_YEAR = 2022;

SELECT T.LIMIT_ADJ_ID,
       T.REC_AD_CODE,
       T.REC_DIV_CODE,
       T.FISCAL_YEAR,
       T.BATCH_NO,
       T.ADJUST_REASON,
       T.AD_NAME,
       T.NEW_SPE_LIMIT_ADJ  AS NEW_SPE_LIMIT_ADJ,
       T.NEW_GEN_LIMIT_ADJ  AS NEW_GEN_LIMIT_ADJ,
       T.FOREIGN_LIMIT_ADJ  AS FOREIGN_LIMIT_ADJ,
       T.NEW_GENDEBT_LIMIT  AS NEW_GENDEBT_LIMIT,
       T.NEW_LIMIT_SUM_ADJ  AS NEW_LIMIT_SUM_ADJ,
       T.NEW_SPEDEBT_LIMIT  AS NEW_SPEDEBT_LIMIT,
       T.FOREIGN_DEBT_LIMIT AS FOREIGN_DEBT_LIMIT,
       T.NEW_LIMIT_SUM      AS NEW_LIMIT_SUM
FROM GD_DEBT_V_LIMIT_ADJ_LIMIT T
WHERE T.DATA_TYPE = '01'
  AND T.IS_DELETED = 2
  AND T.FISCAL_YEAR = 2022
  and exists(select 1
             from dsy_t_wf_task_complete wf
             where wf.wf_id = concat(T.wf_id, '')
               and wf.node_current_id = 1
               and wf.ywsj_id = T.LIMIT_ADJ_ID)
  and T.WF_ID = 100115
  and (T.NODE_CURRENT_ID != 1 or (T.NODE_CURRENT_ID = 1 and T.WF_STATUS != '001'))
  and T.ad_code like concat(42, '%')
  AND T.ad_code like concat(42, '%')
ORDER BY FISCAL_YEAR DESC, BATCH_NO DESC;

债券总限额

SELECT T.FISCAL_YEAR,
       AD.NAME                                                           AS AD_NAME,
       T.BATCH_NO,
       (NVL(T.GENERAL_DEBT_LIMIT, 0) + NVL(T.SPECIAL_DEBT_LIMIT, 0)) / 1 AS HJ,
       NVL(T.GENERAL_DEBT_LIMIT, 0) / 1                                  AS GENERAL_DEBT_LIMIT,
       NVL(T.SPECIAL_DEBT_LIMIT, 0) / 1                                  AS SPECIAL_DEBT_LIMIT
FROM BAS_DEBT_V_LIMIT T
         LEFT JOIN DSY_V_ELE_AD AD ON T.REC_AD_CODE = AD.CODE
WHERE T.BILL_TYPE = '01'
  AND T.DATA_TYPE = '04'
  AND T.REC_AD_CODE = 42
  AND T.FISCAL_YEAR = 2022
ORDER BY T.REC_DATE DESC, T.FISCAL_YEAR DESC, T.BATCH_NO DESC;
